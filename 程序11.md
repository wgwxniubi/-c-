### 1：LU 分解求解线性方程组

```c
// 来源：GeeksforGeeks，《LU decomposition of matrix in C》  
// 链接：https://www.geeksforgeeks.org/lu-decomposition-of-a-matrix-in-c/

#include <stdio.h>
#include <math.h>

#define N 3

void luDecompose(double A[N][N], double L[N][N], double U[N][N]) {
    for (int i = 0; i < N; i++) {
        // Upper Triangular
        for (int k = i; k < N; k++) {
            double sum = 0.0;
            for (int j = 0; j < i; j++)
                sum += (L[i][j] * U[j][k]);
            U[i][k] = A[i][k] - sum;
        }
        // Lower Triangular
        for (int k = i; k < N; k++) {
            if (i == k) L[i][i] = 1.0;
            else {
                double sum = 0.0;
                for (int j = 0; j < i; j++)
                    sum += (L[k][j] * U[j][i]);
                L[k][i] = (A[k][i] - sum) / U[i][i];
            }
        }
    }
}

void forwardSub(double L[N][N], double b[N], double y[N]) {
    for (int i = 0; i < N; i++) {
        double sum = 0.0;
        for (int j = 0; j < i; j++)
            sum += L[i][j] * y[j];
        y[i] = (b[i] - sum) / L[i][i];
    }
}

void backSub(double U[N][N], double y[N], double x[N]) {
    for (int i = N-1; i >= 0; i--) {
        double sum = 0.0;
        for (int j = i+1; j < N; j++)
            sum += U[i][j] * x[j];
        x[i] = (y[i] - sum) / U[i][i];
    }
}

int main() {
    double A[N][N] = {
        {4.0, 3.0, 2.0},
        {3.0, 2.0, 1.0},
        {2.0, 1.0, 3.0}
    };
    double b[N] = {10.0, 7.0, 8.0};
    double L[N][N] = {{0}}, U[N][N] = {{0}}, y[N], x[N];
    luDecompose(A, L, U);
    forwardSub(L, b, y);
    backSub(U, y, x);
    printf("Solution x:\n");
    for (int i = 0; i < N; i++)
        printf("x[%d] = %.6f\n", i, x[i]);
    return 0;
}
```

### 2：2D Laplace 方程 Jacobi 迭代

```c
// 来源：GeeksforGeeks，《Laplace equation solver using Jacobi method》  
// 链接：https://www.geeksforgeeks.org/laplace-equation-solver-using-jacobi-method/

#include <stdio.h>
#include <math.h>

#define M 5
#define N 5
#define MAX_IT 1000
#define TOL 1e-4

void jacobi(double grid[M][N], double newg[M][N]) {
    for (int i = 1; i < M-1; i++)
        for (int j = 1; j < N-1; j++)
            newg[i][j] = 0.25*(grid[i-1][j] + grid[i+1][j]
                             + grid[i][j-1] + grid[i][j+1]);
}

int main() {
    double grid[M][N] = {0}, newg[M][N] = {0};
    // 边界条件
    for (int i = 0; i < M; i++) { grid[i][0] = 100.0; newg[i][0] = 100.0; }
    for (int it = 0; it < MAX_IT; it++) {
        jacobi(grid, newg);
        double diff = 0.0;
        for (int i = 1; i < M-1; i++)
            for (int j = 1; j < N-1; j++) {
                diff = fmax(diff, fabs(newg[i][j] - grid[i][j]));
                grid[i][j] = newg[i][j];
            }
        if (diff < TOL) break;
    }
    printf("Steady-state temperature:\n");
    for (int i = 0; i < M; i++) {
        for (int j = 0; j < N; j++)
            printf("%.2f ", grid[i][j]);
        printf("\n");
    }
    return 0;
}
```

### 3：共轭梯度法求解对称正定方程组

```c
// 来源：GeeksforGeeks，《Conjugate Gradient Method for solving linear equations》  
// 链接：https://www.geeksforgeeks.org/conjugate-gradient-method-solving-linear-equations/

#include <stdio.h>
#include <math.h>

#define N 3
#define MAX_IT 100
#define EPS 1e-6

void matVec(double A[N][N], double x[N], double y[N]) {
    for (int i = 0; i < N; i++) {
        y[i] = 0.0;
        for (int j = 0; j < N; j++)
            y[i] += A[i][j] * x[j];
    }
}

double dot(double x[N], double y[N]) {
    double s = 0.0;
    for (int i = 0; i < N; i++) s += x[i]*y[i];
    return s;
}

void cg(double A[N][N], double b[N], double x[N]) {
    double r[N], p[N], Ap[N];
    matVec(A, x, r);
    for (int i = 0; i < N; i++)
        r[i] = b[i] - r[i], p[i] = r[i];
    double rsold = dot(r, r);
    for (int it = 0; it < MAX_IT; it++) {
        matVec(A, p, Ap);
        double alpha = rsold / dot(p, Ap);
        for (int i = 0; i < N; i++)
            x[i] += alpha * p[i], r[i] -= alpha * Ap[i];
        double rsnew = dot(r, r);
        if (sqrt(rsnew) < EPS) break;
        for (int i = 0; i < N; i++)
            p[i] = r[i] + (rsnew/rsold) * p[i];
        rsold = rsnew;
    }
}

int main() {
    double A[N][N] = {{4.0,1.0,1.0},{1.0,3.0,1.0},{1.0,1.0,2.0}};
    double b[N] = {6.0,6.0,5.0}, x[N] = {0};
    cg(A, b, x);
    printf("Solution x:\n");
    for (int i = 0; i < N; i++)
        printf("x[%d] = %.6f\n", i, x[i]);
    return 0;
}
```

### 4：多项式最小二乘拟合

```c
// 来源：GeeksforGeeks，《Polynomial Curve Fitting (Least Square Method)》  
// 链接：https://www.geeksforgeeks.org/polynomial-curve-fitting/

#include <stdio.h>
#include <math.h>

#define M 5  // 数据点数
#define DEG 2 // 多项式次数

void polyFit(double x[], double y[], int m, int deg, double coef[]) {
    double X[2*DEG+1];
    for (int i = 0; i < 2*deg+1; i++) {
        X[i] = 0.0;
        for (int j = 0; j < m; j++)
            X[i] += pow(x[j], i);
    }
    double B[DEG+1][DEG+2];
    for (int i = 0; i <= deg; i++) {
        for (int j = 0; j <= deg; j++)
            B[i][j] = X[i+j];
        B[i][deg+1] = 0.0;
        for (int j = 0; j < m; j++)
            B[i][deg+1] += pow(x[j], i) * y[j];
    }
    // Gaussian Elimination
    for (int i = 0; i < deg; i++) {
        for (int k = i+1; k <= deg; k++) {
            double t = B[k][i] / B[i][i];
            for (int j = 0; j <= deg+1; j++)
                B[k][j] -= t * B[i][j];
        }
    }
    for (int i = deg; i >= 0; i--) {
        coef[i] = B[i][deg+1];
        for (int j = i+1; j <= deg; j++)
            coef[i] -= B[i][j] * coef[j];
        coef[i] /= B[i][i];
    }
}

int main() {
    double x[M] = {1,2,3,4,5}, y[M] = {5,9,15,23,35};
    double coef[DEG+1];
    polyFit(x, y, M, DEG, coef);
    printf("Fitted polynomial coefficients:\n");
    for (int i = 0; i <= DEG; i++)
        printf("a[%d] = %.6f\n", i, coef[i]);
    return 0;
}
```

### 5：弦截法（False Position）求方程根

```c
// 来源：GeeksforGeeks，《False Position Method to find root of equation》  
// 链接：https://www.geeksforgeeks.org/false-position-method-root-finding/

#include <stdio.h>
#include <math.h>

double f(double x) { return x*x*x + x - 1.0; }

double falsePosition(double a, double b, double tol) {
    double c = a;
    while ((b - a) >= tol) {
        c = (a*f(b) - b*f(a)) / (f(b) - f(a));
        if (f(c) == 0.0) break;
        else if (f(c) * f(a) < 0) b = c;
        else a = c;
    }
    return c;
}

int main() {
    double root = falsePosition(0.0, 1.0, 1e-6);
    printf("Root ≈ %.6f\n", root);
    return 0;
}
```

### 6：Runge–Kutta 四阶法求解阻尼谐振子

```c
// 来源：TutorialsPoint，《Runge Kutta Method in C》  
// 链接：https://www.tutorialspoint.com/runge-kutta-method-in-c

#include <stdio.h>

// y'' + 2*zeta*omega*y' + omega^2*y = 0  转换为一阶系统
void rk4Step(double y[], double dydt[], double zeta, double omega, double h, double yout[], double dyout[]) {
    double k1y = dydt[0];
    double k1v = -2.0*zeta*omega*dydt[0] - omega*omega*y[0];
    double y1 = y[0] + 0.5*h*k1y, v1 = dydt[0] + 0.5*h*k1v;

    double k2y = v1;
    double k2v = -2.0*zeta*omega*v1 - omega*omega*y1;
    double y2 = y[0] + 0.5*h*k2y;   double v2 = dydt[0] + 0.5*h*k2v;

    double k3y = v2;
    double k3v = -2.0*zeta*omega*v2 - omega*omega*y2;
    double y3 = y[0] + h*k3y;       double v3 = dydt[0] + h*k3v;

    double k4y = v3;
    double k4v = -2.0*zeta*omega*v3 - omega*omega*y3;

    yout[0]   = y[0]   + h*(k1y+2*k2y+2*k3y+k4y)/6.0;
    dyout[0] = dydt[0] + h*(k1v+2*k2v+2*k3v+k4v)/6.0;
}

int main() {
    double zeta = 0.1, omega = 2.0*M_PI;
    double h = 0.01;
    int steps = 5;
    double y = 1.0, v = 0.0;
    double yout, vout;
    printf("t\ty\tv\n");
    for (int i = 0; i <= steps; i++) {
        printf("%.2f\t%.6f\t%.6f\n", i*h, y, v);
        rk4Step((double[]){y}, (double[]){v}, zeta, omega, h, &yout, &vout);
        y = yout; v = vout;
    }
    return 0;
}
```

### 7：QR 分解（Gram–Schmidt 正交化）

```c
// 来源：GeeksforGeeks，《QR Decomposition using Gram-Schmidt process》  
// 链接：https://www.geeksforgeeks.org/qr-decomposition-using-gram-schmidt-process/

#include <stdio.h>
#include <math.h>

#define M 3
#define N 3

void qrGS(double A[M][N], double Q[M][N], double R[N][N]) {
    for (int k = 0; k < N; k++) {
        double norm = 0.0;
        for (int i = 0; i < M; i++) norm += A[i][k]*A[i][k];
        R[k][k] = sqrt(norm);
        for (int i = 0; i < M; i++) Q[i][k] = A[i][k] / R[k][k];
        for (int j = k+1; j < N; j++) {
            double dot = 0.0;
            for (int i = 0; i < M; i++) dot += Q[i][k] * A[i][j];
            R[k][j] = dot;
            for (int i = 0; i < M; i++) A[i][j] -= dot * Q[i][k];
        }
    }
}

int main() {
    double A[M][N] = {{12.0,-51.0,4.0},{6.0,167.0,-68.0},{-4.0,24.0,-41.0}};
    double Q[M][N] = {{0}}, R[N][N] = {{0}};
    qrGS(A, Q, R);
    printf("Q Matrix:\n");
    for (int i=0;i<M;i++){
        for(int j=0;j<N;j++) printf("%.4f ", Q[i][j]);
        printf("\n");
    }
    printf("R Matrix:\n");
    for (int i=0;i<N;i++){
        for(int j=0;j<N;j++) printf("%.4f ", R[i][j]);
        printf("\n");
    }
    return 0;
}
```

```c
#include <stdio.h>
#include <math.h>

double erfSimpson(double x, int n) {
    double h = x / n, sum = 0.0;
    for (int i = 0; i <= n; i++) {
        double t = i*h;
        double coeff = (i==0||i==n)?1.0:(i%2==0?2.0:4.0);
        sum += coeff * exp(-t*t);
    }
    return (2.0/sqrt(M_PI)) * sum * h / 3.0;
}
```

```c
void circumcircle(double x1,double y1,double x2,double y2,double x3,double y3,double center[2], double *r) {
```

```c
#include <stdio.h>
#include <math.h>

#define M 3
#define N 3

void dft2D(double in[M][N], double real[M][N], double imag[M][N]) {
    for (int u = 0; u < M; u++) {
        for (int v = 0; v < N; v++) {
            double sumR = 0.0, sumI = 0.0;

        }
```

### 17：三次贝塞尔曲线点计算（Cubic Bézier）

```c
// 来源：GeeksforGeeks，《Program to Draw Cubic Bezier Curve using Double Precision》  
// 链接：https://www.geeksforgeeks.org/c-program-to-calculate-bezier-curves/

#include <stdio.h>

void bezier3(double P[4][2], double t, double B[2]) {
    double u = 1.0 - t;
    // B(t) = u^3 P0 + 3u^2 t P1 + 3u t^2 P2 + t^3 P3
    for (int d = 0; d < 2; d++) {
        B[d] = u*u*u * P[0][d]
             + 3.0*u*u*t * P[1][d]
             + 3.0*u*t*t * P[2][d]
             + t*t*t   * P[3][d];
    }
}

int main() {
    double P[4][2] = {{0.0,0.0},{1.0,2.0},{3.0,3.0},{4.0,0.0}};
    double B[2];
    int steps = 5;
    for (int i = 0; i <= steps; i++) {
```

### 33：Householder 反射 QR 分解

```c
// 来源：GeeksforGeeks，《QR Decomposition Using Householder Reflections》  
// 链接：https://www.geeksforgeeks.org/qr-decomposition-householder-reflection/

#include <stdio.h>
#include <math.h>

#define N 3

void householder(double A[N][N], double Q[N][N], double R[N][N]) {
    // 初始化 R=A, Q=I
    for (int i=0;i<N;i++)
        for (int j=0;j<N;j++){
            R[i][j]=A[i][j];
            Q[i][j]=(i==j); 
        }
    // 只示例一列反射
    double x[3] = {R[0][0], R[1][0], R[2][0]};
    double normx = sqrt(x[0]*x[0]+x[1]*x[1]+x[2]*x[2]);
    double v[3] = {x[0]+normx, x[1], x[2]};
    double normv = sqrt(v[0]*v[0]+v[1]*v[1]+v[2]*v[2]);
    for (int i=0;i<3;i++) v[i]/=normv;
    // H = I - 2vv^T, 更新 R=HR, Q=QH
    for (int i=0;i<N;i++)
        for (int j=0;j<N;j++){
            double Hij = (i==j) ? 1.0 - 2*v[i]*v[j] : -2*v[i]*v[j];
            // 这里仅演示 R 的第一步反射
            R[i][j] = Hij*R[0][j] + ((i>0)?R[i][j]:0);
            Q[i][j] = Hij*Q[i][j];
        }
}

int main() {
    double A[N][N] = {{12.0,-51.0,4.0},{6.0,167.0,-68.0},{-4.0,24.0,-41.0}};
    double Q[N][N], R[N][N];
    householder(A, Q, R);
```

### 36：SOR 迭代求解 2D Laplace 方程

```c
// 来源：GeeksforGeeks，《Successive Over-Relaxation (SOR) method for Laplace’s equation》  
// 链接：https://www.geeksforgeeks.org/successive-over-relaxation-method-for-laplaces-equation/

#include <math.h>

#define N 5
#define MAX_IT 10000
#define OMEGA 1.5
#define TOL 1e-4

void sor(double grid[N][N]) {
    for (int it = 0; it < MAX_IT; it++) {
        double diff = 0.0;
        for (int i = 1; i < N-1; i++) {
            for (int j = 1; j < N-1; j++) {

            }

}
```

### 39：Romberg 积分

```c
// 来源：GeeksforGeeks，《Romberg Integration》  
// 链接：https://www.geeksforgeeks.org/program-romberg-integration-c/

#include <stdio.h>
#include <math.h>

#define MAX 5

double f(double x) { return log(x); }

double romberg(double a, double b) {
    double R[MAX][MAX];
    for (int i = 0; i < MAX; i++) {
        int n = 1 << i;
        double h = (b - a) / n, sum = 0.0;
        if (i == 0) R[i][0] = 0.5 * (f(a) + f(b)) * (b - a);
        else {
            for (int k = 1; k < n; k += 2)
                sum += f(a + k*h);
            R[i][0] = 0.5 * R[i-1][0] + sum * h;
        }
        for (int j = 1; j <= i; j++)
            R[i][j] = R[i][j-1] + (R[i][j-1] - R[i-1][j-1]) / (pow(4.0, j) - 1.0);
    }
    return R[MAX-1][MAX-1];
}

int main() {
    double I = romberg(1.0, 2.0);
    printf("∫₁² ln(x) dx ≈ %.6f\n", I);
```

```c
// 链接：https://www.geeksforgeeks.org/qr-decomposition-using-givens-rotation/

#include <stdio.h>
#include <math.h>

#define N 3

void givensQR(double A[N][N], double Q[N][N], double R[N][N]) {
    for (int i=0;i<N;i++) for(int j=0;j<N;j++){ R[i][j]=A[i][j]; Q[i][j]=(i==j); }
    for (int j = 0; j < N; j++) {
        for (int i = j+1; i < N; i++) {
            double r = hypot(R[j][j], R[i][j]);
            double c = R[j][j]/r, s = -R[i][j]/r;
            for (int k = 0; k < N; k++) {
                double t1 = c*R[j][k] - s*R[i][k];
                double t2 = s*R[j][k] + c*R[i][k];
                R[j][k] = t1; R[i][k] = t2;
                t1 = c*Q[j][k] - s*Q[i][k];
                t2 = s*Q[j][k] + c*Q[i][k];
                Q[j][k] = t1; Q[i][k] = t2;
            }
        }
    }
}
```

```c
#include <math.h>

// 计算某节点期权价值
double binomialOption(double S0, double K, double T, double r, double u, double d, int N) {
    double p = (exp(r*T/N) - d) / (u - d);
    double dt = T / N;
    double option[N+1];
    // 末端价值
    for (int i = 0; i <= N; i++) {
        double ST = S0 * pow(u, N-i) * pow(d, i);
        option[i] = fmax(ST - K, 0.0);

    // 回溯

        for (int i = 0; i < step; i++) {
            option[i] = exp(-r*dt) * (p * option[i] + (1.0 - p) * option[i+1]);
        }
    }
    return option[0];
}

int main() {
    double price = binomialOption(100.0, 100.0, 1.0, 0.05, 1.1, 0.9, 50);
```

```c
    double h[n], alpha[n], l[n+1], mu[n], z[n+1];
```

```c
                u[i][j] = 0.25*(u[i-1][j] + u[i+1][j] + u[i][j-1] + u[i][j+1] - 1.0); 
                diff = fmax(diff, fabs(u[i][j] - old));
            }
        }
        if (diff < TOL) break;
    }
}
```
