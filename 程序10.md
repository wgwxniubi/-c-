### 1：Cramer 法则解三元线性方程组

```c
// 来源：GeeksforGeeks，《Solve a system of linear equations using Cramer’s rule》  
// 链接：https://www.geeksforgeeks.org/solve-a-system-of-linear-equations-using-cramers-rule/

#include <stdio.h>

double determinant3(double M[3][3]) {
    return M[0][0]*(M[1][1]*M[2][2]-M[1][2]*M[2][1])
         - M[0][1]*(M[1][0]*M[2][2]-M[1][2]*M[2][0])
         + M[0][2]*(M[1][0]*M[2][1]-M[1][1]*M[2][0]);
}

void cramersRule(double A[3][3], double B[3], double X[3]) {
    double D = determinant3(A);
    double M[3][3];
    for (int i = 0; i < 3; i++) {
        // 构造替换矩阵
        for (int r = 0; r < 3; r++) for (int c = 0; c < 3; c++)
            M[r][c] = (c==i ? B[r] : A[r][c]);
        X[i] = determinant3(M) / D;
    }
}

int main() {
    double A[3][3] = {{2.0, -1.0, 3.0}, {1.0, 0.0, -2.0}, {3.0, 1.0, 1.0}};
    double B[3] = {5.0, -4.0, 6.0};
    double X[3];
    cramersRule(A, B, X);
    printf("方程组解：x=%.4f, y=%.4f, z=%.4f\n", X[0], X[1], X[2]);
    return 0;
}
```

```c
void eigen2(double M[2][2], double *lambda1, double *lambda2) {
```

```c
// 来源：GeeksforGeeks，《Simpson’s Rule for Numerical Integration》

#include <stdio.h>
#include <math.h>

double f(double x) { return exp(-x*x); }

double simpson(double a, double b, int n) {
    double h = (b - a) / n, sum = f(a) + f(b);
```

### 9：高斯–赛德尔法（Gauss–Seidel）求解线性方程组

```c
// 来源：GeeksforGeeks，《Gauss Seidel method to solve system of linear equations》  
// 链接：https://www.geeksforgeeks.org/gauss-seidel-method-to-solve-system-of-linear-equations/

#include <stdio.h>
#include <math.h>

#define N 3
#define EPS 1e-6
#define MAX_IT 100

void gaussSeidel(double A[N][N], double b[N], double x[N]) {
    for (int i = 0; i < N; i++) x[i] = 0.0;
    for (int it = 0; it < MAX_IT; it++) {
        double maxDiff = 0.0;
        for (int i = 0; i < N; i++) {
            double sum = b[i];
            for (int j = 0; j < N; j++)
                if (j != i) sum -= A[i][j] * x[j];
            double x_new = sum / A[i][i];
            maxDiff = fmax(maxDiff, fabs(x_new - x[i]));
            x[i] = x_new;
        }
        if (maxDiff < EPS) break;
    }
}

int main() {
    double A[N][N] = {{4,1,2},{3,5,1},{1,1,3}};
    double b[N] = {4,7,3};
    double x[N];
    gaussSeidel(A,b,x);
    for (int i = 0; i < N; i++)
        printf("x[%d] = %.6f\n", i, x[i]);
    return 0;
}
```

### 10：Gamma 函数（Lanczos 近似）

```c
// 来源：“Numerical Recipes in C”，Lanczos approximation for Gamma function  
// 链接：https://en.wikipedia.org/wiki/Lanczos_approximation

#include <stdio.h>
#include <math.h>

// Lanczos 参数
static const double lanczos_g = 7;
static const double lanczos_coefs[9] = {
    0.99999999999980993,
    676.5203681218851,
   -1259.1392167224028,
    771.32342877765313,
   -176.61502916214059,
    12.507343278686905,
   -0.13857109526572012,
    9.9843695780195716e-6,

};

double gammaLanczos(double z) {
    double x = lanczos_coefs[0];
```

### 11：Beta 函数 via Gamma

```c

#include <stdio.h>
#include <math.h>

// 复用前述 Lanczos Gamma
double gammaLanczos(double z);

double betaFunc(double x, double y) {
    return gammaLanczos(x) * gammaLanczos(y) / gammaLanczos(x + y);
}

int main() {
    double pairs[][2] = {{0.5,0.5},{2.0,3.0},{5.5,1.5},{3.0,4.0},{1.2,2.8}};

    return 0;

// Lanczos 实现（同上）
double gammaLanczos(double z) {
    static const double g = 7;
    static const double coefs[9] = {
        0.99999999999980993,
        676.5203681218851,
       -1259.1392167224028,
        771.32342877765313,
       -176.61502916214059,
        12.507343278686905,
       -0.13857109526572012,
        9.9843695780195716e-6,
        1.5056327351493116e-7
```

```c
#include <stdio.h>

double logisticStep(double y, double r, double K) {
    return r * y * (1.0 - y / K);

void eulerLogistic(double y0, double r, double K, double t0, double t1, int steps, double y[]) {
```

### 13：Lotka–Volterra 捕食者—猎物模型（RK4 方法）

```c
// 来源：Wikipedia，《Lotka–Volterra equations》  
// 链接：https://en.wikipedia.org/wiki/Lotka–Volterra_equations

#include <stdio.h>

void rk4Step(double x[], double t, double h, double a, double b, double c, double d, double xout[]) {
    double k1[2], k2[2], k3[2], k4[2], xt[2];
    // f1 = a x - b x y, f2 = -c y + d x y
    k1[0] = a*x[0] - b*x[0]*x[1];
    k1[1] = -c*x[1] + d*x[0]*x[1];
    for (int i=0;i<2;i++) xt[i]=x[i]+0.5*h*k1[i];
    k2[0] = a*xt[0] - b*xt[0]*xt[1];
    k2[1] = -c*xt[1] + d*xt[0]*xt[1];
    for (int i=0;i<2;i++) xt[i]=x[i]+0.5*h*k2[i];
    k3[0] = a*xt[0] - b*xt[0]*xt[1];
    k3[1] = -c*xt[1] + d*xt[0]*xt[1];
    for (int i=0;i<2;i++) xt[i]=x[i]+h*k3[i];
    k4[0] = a*xt[0] - b*xt[0]*xt[1];
    k4[1] = -c*xt[1] + d*xt[0]*xt[1];
    for (int i=0;i<2;i++)
        xout[i] = x[i] + h*(k1[i]+2*k2[i]+2*k3[i]+k4[i])/6.0;
}

int main() {
    double a=1.1, b=0.4, c=0.4, d=0.1;
    double x[2] = {10.0, 5.0}, t=0.0, h=0.1;
    for (int i=0; i<20; i++) {
        rk4Step(x, t, h, a,b,c,d, x);
        t += h;
        printf("t=%.1f, prey=%.4f, pred=%.4f\n", t, x[0], x[1]);
    }
    return 0;
}
```

### 18：SIR 疫情模型（Euler 方法）

```c
// 来源：GeeksforGeeks，《Simulation of SIR model using Euler method》  
// 链接：https://www.geeksforgeeks.org/simulation-of-sir-model-in-c-program/

#include <stdio.h>

void sirEuler(double S[], double I[], double R[], int n, double beta, double gamma, double dt) {
    for (int i=1; i<n; i++) {
        double dS = -beta*S[i-1]*I[i-1];
        double dI = beta*S[i-1]*I[i-1] - gamma*I[i-1];
        double dR = gamma*I[i-1];
        S[i] = S[i-1] + dS*dt;
        I[i] = I[i-1] + dI*dt;
        R[i] = R[i-1] + dR*dt;
    }
}

int main() {

    S[0]=0.99; I[0]=0.01; R[0]=0.0;
```

```c
// 来源：GeeksforGeeks，《Composite Simpson's rule for approximate value of definite integral》  
// 链接：https://www.geeksforgeeks.org/composite-simpsons-rule-for-approximate-value-of-definite-integral/

#include <stdio.h>
#include <math.h>

double f(double x) {
    return (x == 0.0) ? 1.0 : sin(x)/x;

double compositeSimpson(double a, double b, int n) {
    double h = (b - a) / n, sum = f(a) + f(b);

    }
    return sum * h / 3.0;
}
```

```c
void thomas(int n, double a[], double b[], double c[], double d[], double x[]) {
    double c_prime[n];
    double d_prime[n];
    c_prime[0] = c[0] / b[0];
    d_prime[0] = d[0] / b[0];
    for (int i = 1; i < n; i++) {
        double m = b[i] - a[i] * c_prime[i-1];
        c_prime[i] = c[i] / m;
        d_prime[i] = (d[i] - a[i] * d_prime[i-1]) / m;
    }
    x[n-1] = d_prime[n-1];
    for (int i = n-2; i >= 0; i--)
        x[i] = d_prime[i] - c_prime[i] * x[i+1];
}

int main() {
    int n = 4;
    double a[] = {0.0, 1.0, 1.0, 1.0}; // 下对角
    double b[] = {4.0, 4.0, 4.0, 4.0}; // 主对角
    double c[] = {1.0, 1.0, 1.0, 0.0}; // 上对角
    double d[] = {5.0, 6.0, 6.0, 5.0}; // 常数项
    double x[4];
    thomas(n, a, b, c, d, x);
    for (int i = 0; i < n; i++)
        printf("x[%d] = %.4f\n", i, x[i]);
    return 0;
}
```

### 23：Cholesky 分解求解对称正定方程组

```c
// 来源：GeeksforGeeks，《Cholesky Decomposition》  
// 链接：https://www.geeksforgeeks.org/cholesky-decomposition-matrix-inversion/

#include <stdio.h>
#include <math.h>

int cholesky(int n, double A[][n], double L[][n]) {
    for (int i = 0; i < n; i++) {
        for (int j = 0; j <= i; j++) {
            double sum = 0.0;
            for (int k = 0; k < j; k++)
                sum += L[i][k] * L[j][k];
            if (i == j)
                L[i][j] = sqrt(A[i][i] - sum);
            else
                L[i][j] = (1.0 / L[j][j] * (A[i][j] - sum));
        }
        for (int j = i+1; j < n; j++)
            L[i][j] = 0.0;
    }
    return 1;
}

void forwardSub(int n, double L[][n], double b[], double y[]) {
    for (int i = 0; i < n; i++) {
        double sum = 0.0;
        for (int k = 0; k < i; k++) sum += L[i][k] * y[k];
        y[i] = (b[i] - sum) / L[i][i];
    }
}

void backwardSub(int n, double L[][n], double y[], double x[]) {
    for (int i = n-1; i >= 0; i--) {
        double sum = 0.0;
        for (int k = i+1; k < n; k++) sum += L[k][i] * x[k];
        x[i] = (y[i] - sum) / L[i][i];
    }
}

int main() {
    int n = 3;
    double A[3][3] = {{25.0, 15.0, -5.0},
                      {15.0, 18.0,  0.0},
                      {-5.0, 0.0, 11.0}};
    double b[] = {350.0, 400.0, 200.0};
    double L[3][3], y[3], x[3];
    cholesky(n, A, L);
    forwardSub(n, L, b, y);
    backwardSub(n, L, y, x);
    for (int i = 0; i < n; i++)
        printf("x[%d] = %.4f\n", i, x[i]);
    return 0;
}
```

### 46：复合中点法数值积分

```c
// 来源：GeeksforGeeks，《Midpoint Rule for numerical integration》  
// 链接：https://www.geeksforgeeks.org/program-mid-point-rule-integration/

#include <stdio.h>

double f(double x) {
    return x*x + x + 1.0;
}

double midpointRule(double a, double b, int n) {
    double h = (b - a) / n, sum = 0.0;
    for (int i = 0; i < n; i++) {
        double xi = a + (i + 0.5)*h;
        sum += f(xi);
    }
    return sum * h;
}

int main() {
    double integral = midpointRule(0.0, 2.0, 1000);
    printf("Midpoint ∫₀² (x²+x+1) dx ≈ %.6f\n", integral);
    return 0;
}
```

### 47：Cauchy 分布概率密度函数（PDF）

```c
// 来源：GeeksforGeeks，《C program for Cauchy distribution》  
// 链接：https://www.geeksforgeeks.org/program-for-cauchy-distribution/

#include <stdio.h>
#include <math.h>

double cauchyPDF(double x, double x0, double gamma) {
    return (1.0/M_PI) * (gamma / ( (x-x0)*(x-x0) + gamma*gamma ));
}

int main() {
    double x[] = {-2.0, -1.0, 0.0, 1.0, 2.0};
    int n = sizeof(x)/sizeof(x[0]);
    double x0 = 0.0, gamma = 1.0;
    for (int i = 0; i < n; i++) {
        double y = cauchyPDF(x[i], x0, gamma);
        printf("CauchyPDF(%.1f) = %.6f\n", x[i], y);
    }
    return 0;
}
```

### 48：Hyperbolic sine (sinh) 的 Taylor 级数展开

```c
// 来源：Wikipedia，《Hyperbolic sine》  
// 链接：https://en.wikipedia.org/wiki/Hyperbolic_sine#Series

#include <stdio.h>
#include <math.h>

// 计算 sinh(x)≈∑_{n=0 to terms-1} x^(2n+1)/(2n+1)!
double factorial(int n) {
    double f = 1.0;
    for (int i = 2; i <= n; i++) f *= i;
    return f;
}

double taylorSinh(double x, int terms) {
    double sum = 0.0;
    for (int n = 0; n < terms; n++) {
        sum += pow(x, 2*n+1) / factorial(2*n+1);
    }
    return sum;
}

int main() {
    double xs[] = {0.5, 1.0, 2.0, 3.0};
    int n = sizeof(xs)/sizeof(xs[0]);
    for (int i = 0; i < n; i++) {
        double approx = taylorSinh(xs[i], 10);
        double actual = sinh(xs[i]);
        printf("sinh(%.1f): Taylor≈%.6f, math.h≈%.6f\n", xs[i], approx, actual);
    }
    return 0;
}
```

### 50：Chebyshev 多项式 Tₙ(x) 递归计算

```c
// 来源：Wikipedia，《Chebyshev polynomials》  
// 链接：https://en.wikipedia.org/wiki/Chebyshev_polynomials#Definition

#include <stdio.h>

double chebyshevT(int n, double x) {
    if (n == 0) return 1.0;
    if (n == 1) return x;
    return 2.0 * x * chebyshevT(n-1, x) - chebyshevT(n-2, x);
}

int main() {
    double x = 0.5;
    for (int n = 0; n <= 5; n++) {
        printf("T_%d(%.2f) = %.6f\n", n, x, chebyshevT(n, x));
    }
    return 0;
}
```
