## 1. 热力学能量平衡计算 X

```c++
// 来源：GeeksforGeeks，《C Program to Calculate Enthalpy in Thermodynamics》
// 链接：https://www.geeksforgeeks.org/c-program-to-calculate-enthalpy-in-thermodynamics/
#include <stdio.h>

int main() {
    double m = 2.5; // 质量 kg
    double cp = 4.18; // 比热容 kJ/kg·K
    double t1 = 25.0, t2 = 80.0; // 初末温度 °C
    double deltaT = t2 - t1;
    double q = m * cp * deltaT; // 热量
    printf("吸收热量 Q = %.2f kJ\n", q);
    return 0;
}
```

## 47. 几何平均值（Geometric Mean）

```c
// 来源：GeeksforGeeks，《C program to calculate Geometric Mean》
// 链接：https://www.geeksforgeeks.org/geometric-mean/
#include <stdio.h>
#include <math.h>

double geometricMean(double arr[], int n) {
    double logSum = 0.0;
    for (int i = 0; i < n; i++)
        logSum += log(arr[i]);
    return exp(logSum / n);
}

int main() {
    double data[] = {2.0, 8.0, 4.0, 16.0};
    int size = sizeof(data) / sizeof(data[0]);
    printf("几何平均值 = %.4f\n", geometricMean(data, size));
    return 0;
}
```

## 55. 示例 `float` 与 `double` 精度差异X

```c
// 来源：GeeksforGeeks，《Difference between float and double in C/C++》
// 链接：https://www.geeksforgeeks.org/difference-float-double-c-cpp/
#include <stdio.h>
#include <math.h>

void double_solve(double a, double b, double c) {
    double d = b*b - 4.0*a*c;
    double sd = sqrt(d);
    printf("Double roots: %.5f, %.5f\n", (-b+sd)/(2*a), (-b-sd)/(2*a));
}

void float_solve(float a, float b, float c) {
    float d = b*b - 4.0f*a*c;
    float sd = sqrtf(d);
    printf("Float roots: %.5f, %.5f\n", (-b+sd)/(2*a), (-b-sd)/(2*a));
}

int main() {
    float fa=1.0f, fb=-4.0f, fc=3.9999999f;
    double da=1.0, db=-4.0, dc=3.9999999;
    float_solve(fa, fb, fc);
    double_solve(da, db, dc);
    return 0;
}
```
