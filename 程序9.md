### 1：计算内部收益率（IRR，Newton-Raphson 法）

```c
// 来源：GeeksforGeeks，《Program to calculate Internal Rate of Return (IRR)》  
// 链接：https://www.geeksforgeeks.org/program-to-calculate-internal-rate-of-return-irr/

#include <stdio.h>
#include <math.h>

double npv(double cashflow[], int n, double rate) {
    double sum = 0.0;
    for (int i = 0; i < n; i++)
        sum += cashflow[i] / pow(1.0 + rate, i);
    return sum;
}

double irr(double cashflow[], int n) {
    double rate = 0.1, h;
    for (int iter = 0; iter < 20; iter++) {
        double f = npv(cashflow, n, rate);
        // 数值微分近似
        h = 1e-6;
        double f1 = npv(cashflow, n, rate + h);
        rate = rate - f * h / (f1 - f);
    }
    return rate;
}

int main() {
    double cashflow[] = {-10000.0, 3000.0, 4200.0, 6800.0};
    int n = sizeof(cashflow) / sizeof(cashflow[0]);
    double result = irr(cashflow, n);
    printf("IRR ≈ %.6f\n", result);
    return 0;
}
```

```c
// 链接：https://www.geeksforgeeks.org/program-to-find-arc-length-of-a-curve/

#include <stdio.h>
#include <math.h>

double arcLength(int n) {
    double a = 0.0, b = 1.0, h = (b - a) / n, sum = 0.0;
    for (int i = 0; i <= n; i++) {
        double x = a + i * h;
        double ds = sqrt(1 + (2.0*x)*(2.0*x));
        if (i == 0 || i == n) sum += ds;
        else sum += 2 * ds;
    }
    return (h / 2.0) * sum;
}
```

```c
void trajectory(double v0, double angle, double t[], double x[], double y[], int n, double g) {
```

### 17：Black‑Scholes 期权定价（Black‑Scholes Formula）

```c
// 来源：GeeksforGeeks，《C Program for Black Scholes Option Pricing Model》  
// 链接：https://www.geeksforgeeks.org/c-program-for-black-scholes-model-option-pricing/

#include <stdio.h>
#include <math.h>

double normCDF(double x) {
    return 0.5 * (1.0 + erf(x / sqrt(2.0)));
}

double blackScholes(double S, double K, double T, double r, double sigma, int isCall) {
    double d1 = (log(S/K) + (r + 0.5*sigma*sigma)*T) / (sigma*sqrt(T));
    double d2 = d1 - sigma*sqrt(T);
    if (isCall)
        return S*normCDF(d1) - K*exp(-r*T)*normCDF(d2);
    else
        return K*exp(-r*T)*normCDF(-d2) - S*normCDF(-d1);
}

int main() {
    double S = 100.0, K = 100.0, T = 1.0, r = 0.05, sigma = 0.2;
    printf("Call Price = %.4f\n", blackScholes(S,K,T,r,sigma,1));
    printf("Put Price  = %.4f\n", blackScholes(S,K,T,r,sigma,0));
    return 0;
}
```

### 23：幂法求矩阵最大特征值（Power Method）

```c
// 来源：GeeksforGeeks，《Power method to find dominant eigenvalue of a matrix》  
// 链接：https://www.geeksforgeeks.org/power-method-to-find-largest-eigenvalue-and-its-corresponding-eigenvector/

#include <stdio.h>
#include <math.h>

#define N 3
#define ITER 20

void matVecMul(double A[N][N], double x[], double y[]) {
    for (int i = 0; i < N; i++) {
        y[i] = 0.0;
        for (int j = 0; j < N; j++)
            y[i] += A[i][j] * x[j];
    }
}

double powerMethod(double A[N][N], double x[]) {
    double y[N], lambda = 0.0;
    for (int it = 0; it < ITER; it++) {
        matVecMul(A, x, y);
        lambda = fabs(y[0]);
        for (int i = 1; i < N; i++)
            if (fabs(y[i]) > lambda) lambda = fabs(y[i]);
        for (int i = 0; i < N; i++)
            x[i] = y[i] / lambda;
    }
    return lambda;
}

int main() {
    double A[N][N] = {
        {4.0, 1.0, 1.0},
        {1.0, 3.0, 0.0},
        {1.0, 0.0, 2.0}
    };
    double x[N] = {1.0, 1.0, 1.0};  // 初始向量
    double dominant = powerMethod(A, x);
    printf("最大特征值 ≈ %.6f\n", dominant);
    return 0;
```

### 25：多边形面积（Shoelace 公式）

```c
// 来源：GeeksforGeeks，《Find area of a polygon using Shoelace formula》  
// 链接：https://www.geeksforgeeks.org/program-shoelace-formula-find-area-given-polygon/

#include <stdio.h>

double polygonArea(double X[], double Y[], int n) {
    double area = 0.0;
    for (int i = 0; i < n; i++) {
        int j = (i + 1) % n;
        area += X[i] * Y[j] - X[j] * Y[i];
```

### 38：微分方程稳态解迭代（Finite Difference Poisson 1D）

```c
// 来源：Numerical Recipes，Poisson Equation Example  
// 链接：https://en.wikipedia.org/wiki/Poisson_equation

#include <stdio.h>
#include <math.h>

#define N 10
#define TOL 1e-6

void poisson1D(double u[], double f[], int n, double dx) {
    for (int iter = 0; iter < 10000; iter++) {
        double err = 0.0;
        for (int i = 1; i < n-1; i++) {
            double u_new = 0.5*(u[i-1] + u[i+1] - dx*dx*f[i]);
            err = fmax(err, fabs(u_new - u[i]));
            u[i] = u_new;
        }
        if (err < TOL) break;
    }
}

int main() {
    double u[N] = {0}, f[N];
    int n = N;
    double dx = 1.0/(n-1);
    for (int i = 0; i < n; i++)
        f[i] = sin(M_PI * i * dx);
    poisson1D(u, f, n, dx);
    printf("Steady-state u:\n");
    for (int i = 0; i < n; i++)
        printf("%.6f ", u[i]);
    printf("\n");
    return 0;
}
```
