## 1. 天体轨道周期计算（NASA公式）

```c
// 来源：NASA Glenn Research Center，《Orbital Period Calculation》
// 链接：https://www.grc.nasa.gov/www/k-12/rocket/satrocket.html
#include <stdio.h>
#include <math.h>
#ifndef M_PI
#define M_PI 3.14159265358979323846
#endif

int main() {
    const double G = 6.67430e-11;   // 引力常数 m³/kg/s²
    const double M_earth = 5.972e24; // 地球质量 kg
    double h = 400000.0;            // 轨道高度 m
    double R_earth = 6371000.0;     // 地球半径 m
    double a = R_earth + h;         // 半长轴 m
    double T = 2 * M_PI * sqrt(pow(a, 3) / (G * M_earth)); // 轨道周期 s
    
    printf("轨道周期: %.2f 秒\n", T);
    return 0;
}
```

## 3. 电容充放电过程计算

```c
// 来源：Electronics Tutorials，《RC Charging Circuit》
// 链接：https://www.electronics-tutorials.ws/rc/rc_1.html
#include <stdio.h>
#include <math.h>

    double V0 = 12.0;     // 电源电压 V
    double R = 1000.0;    // 电阻 Ω
    double C = 0.001;     // 电容 F
    double t = 1.0;       // 时间 s
    double tau = R * C;   // 时间常数 s

    printf("电容电压: %.4f V\n", Vc);
    printf("电流: %.4f A\n", I);
    return 0;
}
```

## 41. 抛物线运动轨迹计算（带空气阻力）

```c
// 来源：NASA Glenn Research Center，《Projectile Motion with Air Resistance》
// 链接：https://www.grc.nasa.gov/www/k-12/airplane/trajair.html
#include <stdio.h>
#include <math.h>
#ifndef M_PI
#define M_PI 3.14159265358979323846
#endif

int main() {
    double v0 = 20.0;    // 初速度 m/s
    double theta = 45.0;  // 发射角度 °
    double g = 9.81;     // 重力加速度 m/s²
    double k = 0.01;     // 空气阻力系数
    double m = 0.1;      // 质量 kg
    double t = 2.0;      // 时间 s
    double rad = theta * M_PI / 180.0;
    double vx = v0 * cos(rad) * exp(-k * t / m);
    double vy = (v0 * sin(rad) - (m * g) / k) * exp(-k * t / m) + (m * g) / k;
    double x = (m / k) * v0 * cos(rad) * (1 - exp(-k * t / m));
    double y = (m / k) * (v0 * sin(rad) + (m * g) / k) * (1 - exp(-k * t / m)) - (m * g * t) / k;
    
    printf("位置: (%.4f, %.4f) m\n", x, y);
    printf("速度: (%.4f, %.4f) m/s\n", vx, vy);
    return 0;
}
```
