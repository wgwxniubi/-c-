```c
#include <stdio.h>

double myPow(double x, int n) {
    if (n == 0) return 1.0;
    double half = myPow(x, n / 2);
    double result = half * half;
    return (n % 2 == 0) ? result
         : (n > 0 ? result * x : result / x);
}

int main() {
    double x = 2.0, ans;
    int n = -3;
    ans = myPow(x, n);              // 递归调用
    printf("myPow(%.2f, %d) = %.5f\n", x, n, ans);

}
```

### 2：LeetCode 69. Sqrt(x)（二分法求解）

```c
// 来源：LeetCode
// 链接：https://leetcode.com/problems/sqrtx/
#include <stdio.h>

double mySqrt(double x) {
    double l = 0.0, r = x, mid, res = 0.0;
    while (r - l > 1e-6) {
        mid = (l + r) / 2.0;
        if (mid * mid <= x) {
            res = mid;
            l = mid;
        } else {
            r = mid;
        }
    }
    return res;
}

int main() {
    double num = 10.0, root;
    root = mySqrt(num);
    printf("sqrt(%.2f) ≈ %.6f\n", num, root);
    return 0;
}
```

### 18：立方根二分查找法（Cube Root via Binary Search）

```c
// 来源：GeeksforGeeks，《C program to find cube root of a number using binary search》  
// 链接：https://www.geeksforgeeks.org/c-program-to-find-cube-root-of-a-number-using-binary-search/

#include <stdio.h>

double cubeRoot(double x) {
    double low = 0.0, high = x > 1 ? x : 1.0, mid;
    while (high - low > 1e-6) {
        mid = (low + high) / 2.0;
        if (mid * mid * mid < x)
            low = mid;
        else
            high = mid;
    }
    return (low + high) / 2.0;
}

int main() {
    double num = 27.0;
    printf("Cube root of %.2f ≈ %.6f\n", num, cubeRoot(num));
    return 0;
}
```

### 23：Gauss–Seidel 迭代法求线性方程组

```c
// 来源：GeeksforGeeks，《Gauss Seidel method to solve system of linear equations》  
// 链接：https://www.geeksforgeeks.org/gauss-seidel-method-to-solve-system-of-linear-equations/

#include <stdio.h>
#include <math.h>

#define N 3
#define EPS 1e-6
#define MAX_IT 100

void gaussSeidel(double A[N][N], double b[N], double x[N]) {
    for (int i = 0; i < N; i++) x[i] = 0.0;
    for (int it = 0; it < MAX_IT; it++) {
        double maxDiff = 0.0;
        for (int i = 0; i < N; i++) {
            double sum = b[i];
            for (int j = 0; j < N; j++) if (j != i)
                sum -= A[i][j] * x[j];
            double x_new = sum / A[i][i];
            maxDiff = fmax(maxDiff, fabs(x_new - x[i]));
            x[i] = x_new;
        }
        if (maxDiff < EPS) break;
    }
}

int main() {
    double A[N][N] = {{4, -1, 0}, {-1, 4, -1}, {0, -1, 3}};
    double b[N] = {15, 10, 10};
    double x[N];
    gaussSeidel(A, b, x);
    for (int i = 0; i < N; i++)
        printf("x[%d] = %.6f\n", i, x[i]);
    return 0;
}
```

### 24：Jacobi 迭代法求线性方程组

```c
// 来源：GeeksforGeeks，《Jacobi method to solve system of linear equations》  
// 链接：https://www.geeksforgeeks.org/jacobi-iteration-method-in-c/

#include <stdio.h>
#include <math.h>

#define M 3
#define TOL 1e-6
#define MAX_ITER 100

void jacobi(double A[M][M], double b[M], double x[M]) {
    double x_old[M];
    for (int i = 0; i < M; i++) x[i] = 0.0;
    for (int it = 0; it < MAX_ITER; it++) {
        for (int i = 0; i < M; i++) x_old[i] = x[i];
        for (int i = 0; i < M; i++) {
            double sum = b[i];
            for (int j = 0; j < M; j++) if (j != i)
                sum -= A[i][j] * x_old[j];
            x[i] = sum / A[i][i];
        }
        double err = 0.0;
        for (int i = 0; i < M; i++)
            err = fmax(err, fabs(x[i] - x_old[i]));
        if (err < TOL) break;
    }
}

int main() {
    double A[M][M] = {{4,1,2},{3,5,1},{1,1,3}};
    double b[M] = {4,7,3};
    double x[M];
    jacobi(A, b, x);
    for (int i = 0; i < M; i++)
        printf("x[%d] = %.6f\n", i, x[i]);
    return 0;
}
```

```c
double bisection(double a, double b, double tol) {
```

```c
// 链接：https://www.geeksforgeeks.org/newton-raphson-method-find-root-equation/

#include <stdio.h>
#include <math.h>

double f(double x) { return x*x - 612.0; }
double df(double x) { return 2*x; }

double newtonRaphson(double x0, int iter) {
    double x = x0;
    for (int i = 0; i < iter; i++) {
        x = x - f(x)/df(x);
    }
    return x;
```

```c
double gaussLegendre(int iter) {
```

### 43：矩阵求逆（Inverse of a Matrix）

```c
// 来源：GeeksforGeeks，《Program to find inverse of a matrix》  
// 链接：https://www.geeksforgeeks.org/program-to-find-inverse-of-a-matrix/

#include <stdio.h>

#define N 3

void getCofactor(double A[N][N], double temp[N][N], int p, int q) {
    int i = 0, j = 0;
    for (int row = 0; row < N; row++) {
        for (int col = 0; col < N; col++) {
            if (row != p && col != q) {
                temp[i][j++] = A[row][col];
                if (j == N - 1) { j = 0; i++; }
            }
        }
    }
}

double determinant(double A[N][N]) {
    double det = 0.0;
    if (N == 1) return A[0][0];
    double temp[N][N];
    int sign = 1;
    for (int f = 0; f < N; f++) {
        getCofactor(A, temp, 0, f);
        det += sign * A[0][f] * determinant(temp);
        sign = -sign;
    }
    return det;
}

void adjoint(double A[N][N], double adj[N][N]) {
    if (N == 1) { adj[0][0] = 1; return; }
    int sign = 1;
    double temp[N][N];
    for (int i = 0; i < N; i++) {
        for (int j = 0; j < N; j++) {
            getCofactor(A, temp, i, j);
            sign = ((i + j) % 2 == 0) ? 1 : -1;
            adj[j][i] = sign * determinant(temp);
        }
    }
}

int inverse(double A[N][N], double inverse[N][N]) {
    double det = determinant(A);
    if (det == 0) return 0;
    double adj[N][N];
    adjoint(A, adj);
    for (int i = 0; i < N; i++)
        for (int j = 0; j < N; j++)
            inverse[i][j] = adj[i][j] / det;
    return 1;
}

int main() {
    double A[N][N] = {{5, -2, 2}, {-1, 3, -1}, {2, -1, 4}};
    double inv[N][N];
    if (inverse(A, inv)) {
        printf("Inverse matrix:\n");
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < N; j++)
                printf("%.4f ", inv[i][j]);
            printf("\n");
        }
    } else {
        printf("Singular matrix, can't find inverse\n");
    }
    return 0;
}
```

```c
int compare(const void *a, const void *b) {
    double da = *(double*)a, db = *(double*)b;
    return (da > db) - (da < db);
}

int main() {
    double data[] = {10.0, 2.0, 14.0, 7.0, 21.0, 13.0, 5.0};
    int n = sizeof(data)/sizeof(data[0]);
    int d = 1; // 去掉最小和最大各1个
    qsort(data, n, sizeof(double), compare);
    double tm = trimmedMean(data, n, d);
    printf("Trimmed Mean = %.4f\n", tm);
    return 0;
}
```

### 49：频率分布直方图（Floating-Point Histogram）

```c
// 来源：GeeksforGeeks，《C Program to find frequency distribution of floating point numbers》  
// 链接：https://www.geeksforgeeks.org/c-program-to-find-frequency-distribution-of-floating-point-numbers/

#include <stdio.h>
#include <math.h>

void histogram(double arr[], int n, int bins, int freq[]) {
    double min = arr[0], max = arr[0];
    for (int i=1; i<n; i++) {
        if (arr[i]<min) min=arr[i];
        if (arr[i]>max) max=arr[i];
    }
    double range = max - min;
    for (int i=0;i<bins;i++) freq[i]=0;
    for (int i=0;i<n;i++) {
        int idx = floor((arr[i]-min)/range * bins);
        if (idx==bins) idx--;
        freq[idx]++;
    }
}

int main() {
    double data[] = {1.1,2.3,2.7,3.8,4.0,4.5,5.2};
    int n = sizeof(data)/sizeof(data[0]);
    int bins = 4, freq[4];
    histogram(data, n, bins, freq);
    for (int i=0;i<bins;i++)
        printf("Bin %d: %d\n", i, freq[i]);
    return 0;
}
```

### 50：希尔排序（Shell Sort）对浮点数组排序

```c
// 来源：GeeksforGeeks，《Shell Sort》  
// 链接：https://www.geeksforgeeks.org/shellsort/

#include <stdio.h>

void shellSort(double arr[], int n) {
    for (int gap = n/2; gap > 0; gap /= 2) {
        for (int i = gap; i < n; i++) {
            double temp = arr[i];
            int j;
            for (j = i; j >= gap && arr[j-gap] > temp; j -= gap)
                arr[j] = arr[j-gap];
            arr[j] = temp;
        }
    }
}

int main() {
    double data[] = {12.5,3.1,7.7,1.4,9.9,2.2};
    int n = sizeof(data)/sizeof(data[0]);
    shellSort(data, n);
    printf("Sorted array:\n");
    for (int i=0;i<n;i++)
        printf("%.2f ", data[i]);
    printf("\n");
    return 0;
}
```
