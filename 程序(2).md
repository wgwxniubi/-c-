## 17.sin(x) 与 cos(x) 级数计算

```cpp
// 来源：Tutorialspoint，《Calculate Sin and Cos Values in C++》，链接：https://www.tutorialspoint.com/cplusplus-program-to-calculate-the-value-of-sin-x-and-cos-x :contentReference[oaicite:7]{index=7}
#include <iostream>
#include <cmath>
using namespace std;

// 用泰勒级数近似计算 sin(x)
void cal_sin(float n) {
    float acc = 0.0001f, denominator, sinx, sinval;
    n = n * (3.142f / 180.0f);
    float temp = n; sinx = n; sinval = sin(n);
    int i = 1;
    do {
        denominator = 2 * i * (2 * i + 1);
        temp = -temp * n * n / denominator;
        sinx += temp;
        i++;
    } while (acc <= fabs(sinval - sinx));
    cout << sinx;
}

// 用泰勒级数近似计算 cos(x)
void cal_cos(float n) {
    float acc = 0.0001f, temp = 1, denominator, cosx, cosval;
    n = n * (3.142f / 180.0f);
    cosx = temp; cosval = cos(n);
    int i = 1;
    do {
        denominator = 2 * i * (2 * i - 1);
        temp = -temp * n * n / denominator;
        cosx += temp;
        i++;
    } while (acc <= fabs(cosval - cosx));
    cout << cosx;
}

int main() {
    float n = 30;
    cout << "Sin: "; cal_sin(n); cout << "\nCos: "; cal_cos(60);
    return 0;
}
```

## 25. UVa 104 – Arbitrage（货币套汇，Floyd–Warshall）

```cpp
#include <iostream>
#include <stack>
#include <cstring>
using namespace std;
int n, path[22][22][22];
double d[22][22][22];
void solve(){
    const double EPS=1.01;
    for(int l=1;l<n;l++)
      for(int k=0;k<n;k++)
        for(int i=0;i<n;i++)
          for(int j=0;j<n;j++){
            double x = d[i][k][l-1]*d[k][j][0];
            if (d[i][j][l]<x){
                d[i][j][l]=x; path[i][j][l]=k;
            }
          }
    bool found=false;
    for(int l=1;l<n&&!found;l++)
      for(int i=0;i<n;i++)
        if(d[i][i][l]>EPS){
          stack<int> st; st.push(i);
          int cur=i;
          for(int j=l;j>0;j--){
            cur=path[i][cur][j];
            st.push(cur);
          }
          int start=st.top();
          while(!st.empty()){
            cout<<st.top()+1<<" ";
            st.pop();
          }
          cout<<start+1<<"\n";
          found=true;
          break;
        }
    if(!found) cout<<"no arbitrage sequence exists\n";
}

int main(){
    while(cin>>n){
        memset(d,0,sizeof(d));
        for(int i=0;i<n;i++){
            for(int j=0;j<n;j++){
                if(i==j) { d[i][j][0]=1.0; path[i][j][0]=i; }
                else { cin>>d[i][j][0]; path[i][j][0]=i; }
            }
        }
        solve();
    }
    return 0;
}
```

## 27. Simpson 复合公式积分

```cpp
#include <iostream>
#include <cmath>
using namespace std;
double f(double x){ return sin(x); }
double simpson(double a,double b,int n){
    double h=(b-a)/n, s=f(a)+f(b);
    for(int i=1;i<n;i+=2) s+=4*f(a+i*h);
    for(int i=2;i<n;i+=2) s+=2*f(a+i*h);
    return s*h/3.0;
}
int main(){

    return 0;
}
```

## 29. 快速排序（QuickSort）

```cpp
#include <iostream>
using namespace std;
int partition(int a[],int l,int r){
    int p=a[r],i=l-1;
    for(int j=l;j<r;j++)
        if(a[j]<p) swap(a[++i],a[j]);
    swap(a[i+1],a[r]);
    return i+1;
}
void quickSort(int a[],int l,int r){
    if(l<r){
        int m=partition(a,l,r);
        quickSort(a,l,m-1);
        quickSort(a,m+1,r);
    }
}
int main(){
    int a[]={10,7,8,9,1,5},n=6;
    quickSort(a,0,n-1);
    for(int i=0;i<n;i++) cout<<a[i]<<" ";
    return 0;
}
```

## 32. Leibniz 级数求 π

```c
double pearson(int x[], int y[], int n) {
    int sum_x = 0, sum_y = 0, sum_xy = 0;
    int squareSum_x = 0, squareSum_y = 0;
    for (int i = 0; i < n; i++) {
```

```c
        f[i] = f[i - 1] + f[i - 2];
    return f[n];
}
```

```c
// 链接：https://www.geeksforgeeks.org/selection-sort/
#include <stdio.h>

void selectionSort(int arr[], int n) {
    for (int i = 0; i < n-1; i++) {
        int min = i;
        for (int j = i+1; j < n; j++)
            if (arr[j] < arr[min])
                min = j;
        int temp = arr[min];
        arr[min] = arr[i];
        arr[i] = temp;
    }
}
```

## 101. 汉明距离（Hamming Distance）

```c
// 来源：GeeksforGeeks，《Hamming Distance between two integers》  
// 链接：https://www.geeksforgeeks.org/hamming-distance-two-integers/
#include <stdio.h>

int hammingDistance(unsigned int x, unsigned int y) {
    int dist = 0;
    unsigned int val = x ^ y;
    while (val) {
        dist += val & 1;
        val >>= 1;
    }
    return dist;
}

int main() {
    unsigned int a = 25, b = 30;
    printf("Hamming distance between %u and %u is %d\n", a, b, hammingDistance(a, b));
```

```c
int maxSubArraySum(int a[], int size) {
    int max_so_far = a[0], curr_max = a[0];
    for (int i = 1; i < size; i++) {
        curr_max = (a[i] > curr_max + a[i]) ? a[i] : curr_max + a[i];
        max_so_far = (max_so_far > curr_max) ? max_so_far : curr_max;
    }

}
```

## 105. 快速排序（Quick Sort）

```c
// 来源：GeeksforGeeks，《QuickSort Algorithm》  
// 链接：https://www.geeksforgeeks.org/quick-sort/
#include <stdio.h>

void swap(int* a, int* b) {
    int t = *a;
    *a = *b;
    *b = t;
}

int partition(int arr[], int low, int high) {
    int pivot = arr[high]; 
    int i = low - 1;
    for (int j = low; j <= high - 1; j++) {
        if (arr[j] < pivot) {
            i++; 
            swap(&arr[i], &arr[j]);
        }
    }
    swap(&arr[i + 1], &arr[high]);
    return i + 1;
}

void quickSort(int arr[], int low, int high) {
    if (low < high) {
        int pi = partition(arr, low, high);
        quickSort(arr, low, pi - 1);
        quickSort(arr, pi + 1, high);
    }
}

int main() {
    int arr[] = {10, 7, 8, 9, 1, 5};
    int n = sizeof(arr)/sizeof(arr[0]);
    quickSort(arr, 0, n - 1);
    printf("Sorted array: ");
    for (int i = 0; i < n; i++)
        printf("%d ", arr[i]);
    return 0;
}
```
